	def fatorial(x):
		parar = x + 1
		result = 1
		for i in range(0, parar):
			result = result * i

	4 * 3 * 2 * 1 = 24

	**********************************

	expressão de fibonacci

	0 1 1 2 3 5 8 13 ... n

	def fibonacci(x):
		primeiro = 0
		segundo = 0
		result = 0
		for i in range(0, x):
			if segundo == 0:
				print(result)
				segundo += 1
			else:
				if result == 1:
					print(result)

				result = primeiro + segundo
				primeiro = segundo
				segundo = result
				print(result)

	com recursividade

	fibonacci(num) = num, se num <= 1, e
	fibonacci(num) = fibonacci(num - 1) + fibonacci(num - 2),
	para num > 1.

	def fibonacci(x):
		if x <= 1:
			return x
		else:
			return fibonacci(x -1) + fibonacci(x -2)

	print(fibonacci(x - 1))

	[[ Manipulando arquivos de texto ]] ===========================================================

		Modos de acesso a arquivos com open()

	r  -> somente leitura
	r+ -> leitura e escrita
	w  -> escrita, apagando o conteúdo existe no arquivo
	w+ -> leitura e escrita. O arquivo é criado, se não existir, é truncado. O texto é 
	inserido no início do arquivo
	a  -> escrita, presevando o conteúdo existente. O arquivo é criado se não existir.
	b  -> modo binário
	+  -> atualização - leitura e escrita
	x  -> abre o arquivo para criação exclusiva.

		É possível combinar os módulos

		* Ler o conteúdo de um arquivo de texto

	read() -> retorna o conteúdo de um arquivo como uma única string.

	readline() -> retorna uma linha do texto a cada chamada, na ordem em que aparece um arquivo.

	readlines() -> retorna uma lista de valores de string do arquivo, sendo que cada string 
	corresponde a uma linha do texto.

	ex.:

	manipulador = open('fibonacci.py', 'r')

	print(manipulador.read())
	manipulador.seek(0) # retorna p/ primeira linha

	print(manipulador.readline())

	manipulador.seek(1)

	print(manipulador.readline())

	manipulador.seek(0)

	print(manipulador.readlines())

	# sem os métodos acima

	for linha in manipulador:
		linha = linha.rstrip()
		print(linha)

	manipulador.close()


	* Retornar apenas algumas linhas expecíficas

	palavra = input("Digite uma palavra:')
	try:
		contador = 0
		with open('arquivo.txt', 'r') as arq:
			for linha in arq:
				linha = linha.rstrip()()
				if palavra in linha:
					contador += 1
					print(linha)

		print('Números de linha', contador)
	except Exception as error:
		print("Erro ao abrir arquivo:", type(error))
